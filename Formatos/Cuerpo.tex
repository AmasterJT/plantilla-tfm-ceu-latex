% Estilo sección tipo "Display" (Salto de línea)
\titleformat{\section}[display]
  {\vspace*{0.5cm}\selectfont\bfseries\color{Negro}} % Estilo común (Arial, Negrita, Versalitas)
  {\fontsize{20pt}{20pt}\selectfont Capítulo \thesection}    % <--- TAMAÑO 20pt para la etiqueta
  {0pt}                                                      % Espacio entre líneas
  {\fontsize{24pt}{24pt}\selectfont}                         % <--- TAMAÑO 24pt para el texto del título

\titlespacing{\section}
  {0pt}                  % Margen izquierdo
  {18pt}                 % Espacio superior
  {10pt}                 % Espacio inferior (ajustado para que no pegue al párrafo)

%Estilo subsección
\titleformat{\subsection}{\normalfont\fontsize{16pt}{20}\bfseries}{\thesubsection}{0.3em}{}
\titlespacing\subsection{0pt}{12pt plus 4pt minus 2pt}{0pt plus 2pt minus 2pt}

%Estilo subsubsección
\titlespacing\subsubsection{0pt}{12pt plus 4pt minus 2pt}{0pt plus 2pt minus 2pt}

% Formato corregido: Negrita (\bfseries) y sin cursiva
\titleformat{\subsubsection}
  {\normalfont\fontsize{14pt}{20}\selectfont\bfseries} % <--- Aquí los cambios
  {\thesubsubsection}
  {0.3em}
  {}

\setlength{\belowcaptionskip}{0pt}
